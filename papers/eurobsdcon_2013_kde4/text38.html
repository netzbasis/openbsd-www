KDE4 on OpenBSD: PO_FILES issue

  • Solution:

MODKDE4_post-patch = \

@echo '====> Fixing GETTEXT_PROCESS_PO_FILES() calls'; \

cd ${WRKSRC} && find . -name CMakeLists.txt | sort | \

while read F; do \

perl -pi.pofilesfix -e '\

if (/GETTEXT_PROCESS_PO_FILES/ and \

!/\sPO_FILES/) { \

s@\$$\{_po_files\}@PO_FILES $$&@; \

}' "$$F"; \

if cmp -s "$$F" "$$F".pofilesfix; then \

rm "$$F".pofilesfix; \

else \

echo "$$F" >&2; \

fi; \

done
